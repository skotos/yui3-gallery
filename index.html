galley-textfield-labelled

This module is to insert a textfield label within the frame of the textfield itself. This label then disappears when a value is entered into the field.

Example

Code

<input type="text" id="ubox" label="Username" />
<input type="password" id="pbox" label="Password" />
<script src="http://yui.yahooapis.com/3.4.1/build/yui/yui-min.js"></script>
<script type="text/javascript">
YUI({
  combine:true,
  groups: {
    sp: {
      base: 'build/',
      combine: false,
      modules: {
        "gallery-textfield-labelled": {
          path: 'gallery-textfield-labelled/gallery-textfield-labelled.js',
          requires: ['node', 'widget', 'event', 'event-key'],
          skinnable: true
        }
      }
    }
  }
}).use('gallery-textfield-labelled', function(Y) {
  var uBox = new TextfieldLabelled({
    srcNode: '#ubox',
    label: "Username"
  }).render();
  var pBox = new TextfieldLabelled({
    srcNode: '#pbox',
    label: "Password"
  }).render();
});
</script>
